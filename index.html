🔗 Google Sheets Live Sync One-Click Update
Works with your private sheet — no sharing required. Uses Google Apps Script (built into Google Sheets) to bridge your private sheet to this dashboard. You paste one small script, deploy it once, paste the URL here, and you're done.
ONE-TIME SETUP (5 min)
1
Open Apps Script in your Google Sheet
In the NAM Dashboard sheet: Extensions → Apps Script
2
Paste this script (replace all existing code)
function doGet(e) {
  var secret = PropertiesService.getScriptProperties().getProperty('SECRET') || 'deka-nam-2026';
  var cb = e.parameter.callback || '';

  function respond(obj) {
    var json = JSON.stringify(obj);
    if (cb) {
      return ContentService.createTextOutput(cb + '(' + json + ')')
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return ContentService.createTextOutput(json)
      .setMimeType(ContentService.MimeType.JSON);
  }

  if (e.parameter.key !== secret) return respond({error:'Unauthorized'});

  var ss = SpreadsheetApp.getActiveSpreadsheet();

  if (e.parameter.action === 'list') {
    var tabs = ss.getSheets().map(function(s){ return s.getName(); });
    return respond({tabs: tabs});
  }

  var tabName = e.parameter.tab || '';
  var sheet = tabName ? ss.getSheetByName(tabName) : ss.getSheets()[0];
  if (!sheet) return respond({error: 'Tab not found: ' + tabName});

  var data = sheet.getDataRange().getDisplayValues();
  return respond({tab: sheet.getName(), data: data});
}
